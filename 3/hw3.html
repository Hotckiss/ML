import time

def learn_time(clf, values, labels):
    s = time.time()
    clf.fit(values, labels)
    f = time.time()
    return f - s
In [258]:
def compare(svm, rf, values_train, labels_train, values_test, labels_test):
    svm_time = learn_time(svm, values_train, labels_train)
    svm_score = svm.score(values_test, labels_test)
    rf_time = learn_time(rf, values_train, labels_train)
    rf_score = rf.score(values_test, labels_test)
    print("svm time: " + str(svm_time))
    print("svm acc: " + str(svm_score))
    print("rf time: " + str(rf_time))
    print("rf acc: " + str(rf_score))
    print("Dtime: " + str(svm_time - rf_time))
    print("Dacc: " + str(svm_score - rf_score))

from scipy.stats import norm
from sklearn import metrics

class NBGaussian:
    def __init__(self, values_train, labels_train):
        self.unique_labels, counts = np.unique(labels_train, return_counts=True)
        self.class_probabilities = counts / len(labels_train)
        means, deviations = [], []
        
        for label in self.unique_labels:
            cur_values = values_train[labels_train == label, :]
            means.append(np.mean(cur_values, axis=0))
            deviations.append(np.std(cur_values, axis=0))
        
        self.means = np.array(means)
        self.deviations = np.array(deviations)
        
    def predict(self, values, labels):
        res = np.empty((len(values), len(self.unique_labels)))
        
        for label in range(len(self.unique_labels)):
            cur = np.empty(values.shape)
            
            for i, j in np.ndindex(values.shape):
                cur[i][j] = norm.pdf(values[i][j], loc=self.means[label][j], scale=self.deviations[label][j])
            
            res[:, label] = self.class_probabilities[label] * np.prod(cur, axis=1) 
        
        for i in range(len(res)):
            sum_ = np.sum(res[i])
            res[i] = 0.5 if sum_ == 0 else res[i] / sum_
                
        return res, np.mean(labels == self.unique_labels[np.argmax(res, axis=1)])

def stemm(value):
    return [PorterStemmer().stem(word) for word in word_tokenize(value)]

def stemm_all(values):
        return np.array([word for value in values for word in stemm(value)])

class NBMultinomial:
    def __init__(self, values_train, labels_train):
        self.unique_labels, counts = np.unique(labels_train, return_counts=True)
        self.class_probabilities = counts / len(labels_train)
        unique_words = np.unique(stemm_all(values_train))
        self.word_dict = {unique_words[i]: i for i in range(len(unique_words))}
        
        word_frequency = []
        for label in self.unique_labels:
            cur_words = stemm_all(values_train[labels_train == label])
            word_frequency.append(np.array([np.sum(cur_words == word) for word in unique_words]))
        self.word_frequency = np.array(word_frequency)
        
    def predict(self, values, labels):
        values = np.array([stemm(row) for row in values])
        res = np.empty((len(values), len(self.unique_labels)))
        
        for label in range(len(self.unique_labels)):
            cnt = np.sum(self.word_frequency[label]) + len(self.word_dict)
                
            for i in range(len(values)):
                p = 1
                
                for j in range(len(values[i])):
                    if values[i][j] in self.word_dict:
                        p *= (self.word_frequency[label][self.word_dict[values[i][j]]] + 1) / cnt
                
                res[i][label] = self.class_probabilities[label] * p
        
        for i in range(len(res)):
            sum_ = np.sum(res[i])
            res[i] = 0.5 if sum_ == 0 else res[i] / sum_
            
        return res, np.mean(labels == self.unique_labels[np.argmax(res, axis=1)])
